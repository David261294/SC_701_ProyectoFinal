use Proyecto

create table CITAS
(ID_CITAS INT NOT NULL,
NOMBRE_PACIENTE_CITA VARCHAR (30) NOT NULL,
APELLIDO_PACIENTE_CITA VARCHAR (30) NOT NULL,
DÍA_CITA INT NOT NULL,
MES_CITA VARCHAR (20) NOT NULL,
AÑO_CITA INT NOT NULL,
HORA_CITA INT NOT NULL,
PRIMARY KEY (ID_CITAS))



CREATE TABLE CONTACTO
(ID_CONTACTO INT NOT NULL,
NOMBRE_CONTACTO VARCHAR(30) NOT NULL,
APELLIDO_CONTACTO VARCHAR(30) NOT NULL,
CORREO_CONTACTO VARCHAR(100) NOT NULL,
DIRECCION_CONTACTO VARCHAR(100) NOT NULL,
TELEFONO_CONTACTO VARCHAR(15) NOT NULL,
PRIMARY KEY(ID_CONTACTO))


CREATE TABLE DENTISTA
(ID_DENTISTA INT NOT NULL,
NOMBRE_DENTISTA VARCHAR(30) NOT NULL,
APELLIDO_DENTISTA VARCHAR(30) NOT NULL,
CONTRASEÑA VARCHAR(40) NOT NULL,
PRIMARY KEY(ID_DENTISTA))




CREATE TABLE EXPEDIENTE_PACIENTE
(ID_PACIENTE INT NOT NULL,
CEDULA INT NOT NULL,
NOMBRE_PACIENTE VARCHAR (30) NOT NULL,
APELLIDO_PACIENTE VARCHAR (30) NOT NULL,
FECHA_NACIMIENTO VARCHAR (50) NOT NULL,
EDAD INT NOT NULL,
PADECIMIENTOS VARCHAR (50) NOT NULL, 
PRIMARY KEY (ID_PACIENTE))



CREATE TABLE PACIENTE_NUEVO 
(ID_PACIENTE_NUEVO INT NOT NULL,
NOMBRE_PACIENTE_NUEVO VARCHAR(30) NOT NULL,
APELLIDO_PACIENTE_NUEVO VARCHAR(30) NOT NULL,
CONTRASEÑA VARCHAR(40) NOT NULL,
PRIMARY KEY(ID_PACIENTE_NUEVO))



CREATE TABLE TRATAMIENTO
(ID_TRATAMIENTO INT NOT NULL,
NOMBRE_TRATAMIENTO VARCHAR(30) NOT NULL,
COSTO_TRATAMIENTO VARCHAR(30),
PRIMARY KEY(NOMBRE_TRATAMIENTO))



CREATE TABLE PRODUCTO
(ID_PRODUCTO INT NOT NULL,
NOMBRE_PRODUCTO VARCHAR(40),
PRECIO_PRODUCTO VARCHAR(30),
PRIMARY KEY(ID_PRODUCTO))




CREATE TABLE FACTURA_PACIENTE
(CODIGO_FACTURA INT NOT NULL,
ID_PACIENTE INT NOT NULL,
FECHA_FACTURA VARCHAR (30) NOT NULL,
DESCUENTO VARCHAR(30) NOT NULL,
IVA VARCHAR(30) NOT NULL,
PRIMARY KEY (CODIGO_FACTURA),
FOREIGN KEY (ID_PACIENTE) REFERENCES EXPEDIENTE_PACIENTE(ID_PACIENTE))




CREATE TABLE DETALLE_FACTURA
(CODIGO_DETALLE INT NOT NULL,
CODIGO_FACTURA INT NOT NULL,
ID_TRATAMIENTO INT NOT NULL,
ID_PRODUCTO INT NOT NULL,
CANTIDAD INT NOT NULL,
PRECIO_FINAL VARCHAR(30) NOT NULL,
PRIMARY KEY (CODIGO_DETALLE),
FOREIGN KEY (CODIGO_FACTURA) REFERENCES FACTURA_PACIENTE(CODIGO_FACTURA),
FOREIGN KEY (ID_TRATAMIENTO) REFERENCES TRATAMIENTO(ID_TRATAMIENTO),
FOREIGN KEY (ID_PRODUCTO) REFERENCES PRODUCTO(ID_PRODUCTO))


CREATE TABLE RECEPCIONISTA
(ID_RECEPCIONISTA INT NOT NULL,
NOMBRE_RECEPCIONISTA VARCHAR(30) NOT NULL,
APELLIDO_RECEPCIONISTA VARCHAR(30) NOT NULL,
CONTRASEÑA VARCHAR(40) NOT NULL,
PRIMARY KEY (ID_RECEPCIONISTA))


CREATE TABLE INGRESOS
(ID_INGRESO INT NOT NULL,
FECHA_INGRESO VARCHAR (50) NOT NULL,
CODIGO_FACTURA VARCHAR(10) NOT  NULL,
PRIMARY KEY(ID_INGRESO),
FOREIGN KEY(CODIGO_FACTURA) REFERENCES DETALLE_FACTURA(CODIGO_FACTURA))


CREATE TABLE INVENTARIO
(ID_INVENTARIO INT NOT NULL,
CANTIDAD INT NOT NULL,
DESCRIPCIÓN VARCHAR (70) NOT NULL,
PRIMARY KEY (ID_INVENTARIO))



CREATE TABLE REUNION
(NUMERO_REUNION INT NOT NULL,
DIA VARCHAR (70) NOT NULL,
ID_DENTISTA INT NOT NULL,
ID_RECEPCIONISTA INT NOT NULL,
DETALLES_REUNION VARCHAR(120) NOT NULL,
PRIMARY KEY(NUMERO_REUNION),
FOREIGN KEY(ID_DENTISTA) REFERENCES DENTISTA(ID_DENTISTA),
FOREIGN KEY(ID_RECEPCIONISTA) REFERENCES RECEPCIONISTA(ID_RECEPCIONISTA))

CREATE TABLE REVIEW
(ID_REVIEW INT NOT NULL,
ID_CITAS INT NOT NULL,
NOMBRE_PACIENTE_CITA VARCHAR(30) NOT NULL,
COMENTARIO VARCHAR(300) NOT NULL,
ESTRELLAS VARCHAR(10) NOT NULL,
PRIMARY KEY (ID_REVIEW),
FOREIGN KEY (ID_CITAS) REFERENCES CITAS (ID_CITAS))



CREATE TABLE EVENTOS
(ID_EVENTO INT NOT NULL,
DÍA_EVENTO INT NOT NULL,
MES_EVENTO VARCHAR (20) NOT NULL,
AÑO_EVENTO INT NOT NULL,
DETALLES_EVENTO VARCHAR (120) NOT NULL,PRIMARY KEY (ID_EVENTO))
